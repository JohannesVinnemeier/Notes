\subsection{Existence of Smooth Isotopies}

In this section we proof a few statements on the existence of smooth isotopies. Lemma \ref{constr_diffeo:lem:isotopy_lemma} and Theorem \ref{constr_diffeo:thm:isotopy_extension_theorem} follow \cite{Milnor1965hcobord} quite closely. The first Lemma is a slight modification of Lemma 5.7 in \cite{Milnor1965hcobord}.

\begin{lemma}[Isotopy Lemma]
    \label{constr_diffeo:lem:isotopy_lemma}
    Let $h \colon \R^n \to \R^n$ be an orientation preserving embedding with $h(0) = 0$. Then $h$ is isotopic to the identity $\mathrm{rel}\, \{0\}$.
\end{lemma}
\begin{proof}
    The proof is basically the same as in \cite{Milnor1965hcobord}. Since $h(0) = 0$ we can write 
    \[
    h(x) = x_1h_1(x) + \ldots + x_nh_n(x)
    \] 
    for smooth functions $h_i \colon \R^n \to \R^n$ and then define an isotopy 
    \[
    \frac{1}{t}h(tx) = x_1h_1(tx) + \ldots + x_nh_n(tx), \quad 0\leq t\leq1
    \]
    from the linear map
    \[
        A \colon \R^n \to \R^n, x \mapsto x_1h_1(0) + \ldots + x_nh_n(0)
    \]
    to $h$. Since $\mathrm{det}(A) >0$ and the space of all $n \times n$ matrices with positive determinant is path connected we moreover get an isotopy from $A$ to $id$.
\end{proof}

\noindent Next we proof Theorem 5.8 in \cite{Milnor1965hcobord} and give a slightly stronger formulation of it. The main idea behind the proof is beautifully explained in \cite{Milnor1965hcobord}.

\begin{theorem}[Isotopy Extension Theorem]
    \label{constr_diffeo:thm:isotopy_extension_theorem}
    Let $M \subseteq N$ be a submanifold with or without boundary of the boundaryless manifold $N$ and let $h_t \colon M \to N, 0\leq t \leq 1$ be an isotopy of the inclusion $M \hookrightarrow N$. Given any compact subset $K \subseteq M$ and an open set $U \subseteq N$, such that $h_t(K) \subseteq U$ for all $t \in [0,1]$, there exists an isotopy 
    \[
    \overline{h}_t \colon N \to N, 0\leq t\leq1
    \]
    with the following properties
    \begin{enumerate}[i)]
        \item $\overline{h}_0 = \mathrm{id}_N$
        \item $\overline{h}_t(x) = h_t(x)$ for all $x \in K$ and $t \in [0,1]$
        \item $\overline{h}_t(x) = x$ for all $x \in N\setminus U$ and $t \in [0,1]$
    \end{enumerate}
\end{theorem}

\begin{proof}
    We formulate the proof only for the case $\partial M = \emptyset$, see Remark \ref{constr_diffeo:rem:isotopy_extension_thm_boundary}. Let $V \subseteq M$ be a relatively compact neighbourhood of $K$. Then
    \begin{align*}
        H \colon V \times [0,1] &\to N \times [0,1], \\
                    (x,t) &\mapsto (h_t(x),t)
    \end{align*}
    is an embedding, since the map $M \times [0,1] \to N \times [0,1]$ defined by the same formula is an injective immersion. So $H$ definies a submanifold
    \[
    S := \mathrm{im}(H) \subseteq N \times [0,1]
    \]
    as well as the smooth vector field $X := H_\ast(\frac{\partial}{\partial t})$ on $S$, i.e. for $y = h_t(x)$ we have 
    \[
    X_{(y,t)} := D_{(x,t)}H\left(\frac{\partial}{\partial t}\right) = \left( \frac{\partial h_t}{\partial t}(x), \frac{\partial}{\partial t} \right).
    \]
    We can then extend $X$ to some open neighbourhood $W \subseteq U \times [0,1] \subseteq N \times [0,1]$ of $S$ at first arbitrarily and then set the coordinate of the part pointing in the $\frac{\partial}{\partial t}$ direction equal to $1$. Thus we obtain a smooth vector field $Y$ on $W$ of the form 
    \[
    Y_{(y,t)} = (\tilde{X}_{(y,t)},\frac{\partial}{\partial t})
    \]
    which agrees with $X$ on $S$. Choose a cut-off function 
    \[
    \psi \colon N \times [0,1] \to \R
    \]
    which equals $1$ on $H(K \times [0,1])$ and has its support contained in $W$. Then we consider the smooth vector field $Z$ on $N \times [0,1]$ defined by
    \[
    Z := \psi Y + (1-\psi)\frac{\partial}{\partial t}.
    \]
    The idea now is to define the required isotopy $\overline{h}_t$ using the flow of $Z$. For this we want to assume that $U$ is relatively compact, otherwise we may shrink $U$. The reason for this will become clear shortly, but first let us record the crucial properties of $Z$:
    \begin{enumerate}[i)]
        \item $Z$ is of the form 
            \[
                Z_{(y,t)} = (\tilde{Z}_{(y,t)},\frac{\partial}{\partial t}),
            \]
        \item $Z$ agrees with $X$ on $H(K \times [0,1])$,
        \item $Z = \frac{\partial}{\partial t}$ outside of the compact set $\overline{U} \times [0,1]$.
    \end{enumerate}
    
    We claim that for any $y \in N$ the unique maximal integral curve $\psi_y$ of $Z$ starting at $(y,0) \in N \times [0,1]$ is defined for all $t \in [0,1]$. If $y \notin \overline{U}$ this is clear, since the $\psi_y$ is explicitly given by 
    \[
    \psi_y(t) = (y,t), \quad 0 \leq t \leq 1.
    \] 
    For $y \in \overline{U}$ this follows from the fact, that $\psi_y$ has to be contained in the compact set $\overline{U} \times [0,1]$ for all time, where it is defined (due to uniqueness of integral curves and the case $y \notin \overline{U}$ above). Then using property (i) of $Z$ we see that $\psi_y(t)$ is in fact defined for all $t \in [0,1]$. 
    
    So after extending $Z$ to $N \times \R$ (to avoid dealing with flows on manifolds with boundary) we can consider the flowout of $N \times \{0\}$, which is defined on $N \times [0,1]$ by the preceeding observations and hence defines a level-preserving diffeomorphism 
    \begin{align*}
        \overline{H} \colon N \times [0,1] &\to N \times [0,1], \\
                (y,t) & \mapsto \psi_y(t).
    \end{align*}
    Finally we define the required isotopy $\overline{h}_t$ by 
    \[
    \overline{H}(y,t) = (\overline{h}_t(y),t).
    \]
    It remains to show that $\overline{h}_t$ agrees with $h_t$ on $K$. But this follows from property (ii) of $Z$: for $x \in K$ let $\phi_x(t) := (x,t), \; 0 \leq t \leq 1$ be the integral curve of the vector field $\frac{\partial}{\partial t}$ on $M \times [0,1]$. Then 
    \[
    H(x,t) = H(\phi_x(t)) = \psi_x(t) = \overline{H}(x,t).
    \]
    The theorem follows.
\end{proof}

\begin{remark}
    \label{constr_diffeo:rem:isotopy_extension_thm_boundary}
    If $M$ is a manifold with boundary we can define a smooth isotopy as a map 
    \[
    h \colon M \times [0,1] \to N,
    \]
    which extends to a smooth isotopy
    \[
    \overline{h} \colon M \times \R \to N.
    \]
    Then exactly the same proof can be carried out if $M$ is allowed to have boundary by constructing the required vector fields on $N \times \R$ and then restricting everything back to $N \times [0,1]$. Note that this definition of smooth isotpies does not change the isotopy classes of maps in the case $\partial M = \emptyset$ (cf. Lemma \ref{misc:lem:different_def_smooth_homotopy_equivalence}). Of course using the notion of a manifold with edges we can define smooth isotopies in the same way and the proof of Theorem \ref{constr_diffeo:thm:isotopy_extension_theorem} goes through unchanged.
\end{remark}

Together the Isotopy Lemma and the Isotopy Extension Theorem can be used to extend locally defined diffeomorphisms to the whole manifold as follows: Begin with an orientation preserving embedding $\phi \colon U \to M$ of an open subset $U \subseteq M$. Then - after possibly restricting $\phi$ to a smaller domain - use Lemma \ref{constr_diffeo:lem:isotopy_lemma} to construct an isotopy from $\phi$ to the inclusion $U \hookrightarrow M$. Lastly - after once again restricting the isotopy to a smaller domain - we can use Theorem \ref{constr_diffeo:thm:isotopy_extension_theorem} to extend this isotopy to an an isotopy $\overline{h}_t \colon N \to N$ starting at $\mathrm{id}_N$. Then $h_1 \colon N \to N$ is a diffeomorphism extending $\phi$.

Next we state a few interesting corollaries of Theorem \ref{constr_diffeo:thm:isotopy_extension_theorem}.

\begin{corollary}
    \label{constr_diffeo:cor:transitivity_of_diffeo_groups}
    Let $M$ be a manifold without boundary, $p,q \in M$ and $\gamma \colon [0,1] \to M$ a smooth path from $p$ to $q$. Then there exists a diffeomorphism $\phi \colon M \to M$ with $\phi(p) = q$, which is smoothly isotopic to $\mathrm{id}_M$. Moreover this isotopy can be chosen to be stationary outside a given open neighbourhood $U \subseteq M$ of the image of $\gamma$.
\end{corollary}

\begin{proof}
    The smooth path $\gamma$ can be seen as an isotpy of the inclusion $\{p\} \hookrightarrow M$. Then the statement follows directly from Theorem \ref{constr_diffeo:thm:isotopy_extension_theorem}.
\end{proof}

\begin{remark}
    If $\mathrm{Aut}_{id}(M) \subset \mathrm{Aut}(M)$ denotes the subgroup of diffeomorphisms which are smoothly isotopic to $\mathrm{id}_M$, the above corollary in  particular implies that $\mathrm{Aut}_{id}(M)$ acts transitively on $M$.
\end{remark}

We use Corollary \ref{constr_diffeo:cor:transitivity_of_diffeo_groups} to show a corollary from \cite{guillemin2010differential}.

\begin{corollary}
    Let $M$ be a connected manifold without boundary of dimension at least $2$. Given distinct points $x_1,\ldots,x_k,y_1,\ldots,y_k \in M$, there exists a diffeomorphism
    \[
    \psi \colon M \to M
    \]
    with $\psi(x_i) = y_i$, which is smoothly isotopic to the identity. Moreover, we can choose this isotopy to be compactly supported.
\end{corollary}

\begin{proof}
    We proof the statement inductively. For $k=0$ there is nothing to show. So suppose we already have a diffeomorphism $\psi^\prime \colon M \to M$ smoothly isotopic to the identity, where the isotopy is compactly supported and $\psi^\prime(x_i) = y_i$ for $1 \leq i \leq k-1$. Note that $M - \{y_1,\ldots,y_{k-1}\}$ is connected, so Corollary \ref{constr_diffeo:cor:transitivity_of_diffeo_groups} implies the existence of a diffeomorphism $\phi \colon M \to M$ with $\phi(\psi^\prime(x_k)) = y_k$ and $\phi(y_i) = y_i$ for $1 \leq i \leq k-1$, such that $\phi$ is smoothly isotopy to the identity via a compactly supported isotopy. Now set $\psi := \phi \circ \psi^\prime$ and compose the isotopies to get a compactly supported isotopy from $\psi$ to the identity.
\end{proof}

This next corollary essentially says, that there are many embeddings of small dimensional manifolds into large dimensional manifolds.

\begin{corollary}
    \label{constr_diffeo:cor:embeddings}
    Let $M$ be a connected manifold without boundary and $S$ a manifold with or without boundary, such that there exists an embedding $S \hookrightarrow \R^{\mathrm{dim}(M)}$. Moreover we fix two distinct points $s_0,s_1 \in S$. Then for any two points $p,q \in M$ there exists an embedding $\iota \colon S \hookrightarrow M$, such that $\iota(s_0) = p$ and $\iota(s_1) = q$.
\end{corollary}

\begin{proof}
    By assumption we can find an embedding $\iota^\prime \colon S \hookrightarrow U \subseteq M$ into any coordinate neighbourhood $U$ around $p$, such that $\iota^\prime(s_0) = p$. We may assume $ p \neq q$, so we can find a smooth path from $\iota^\prime(s_1)$ to $q$, that does not hit $p$ (if $M$ is $1$-dimensional we may arrange the embedding $\iota^\prime$ in such a way, that $\iota^\prime(s_1)$ lies in the same connected component of the manifold $M - \{p\}$ as $q$). Then Corollary \ref{constr_diffeo:cor:transitivity_of_diffeo_groups} implies the existence of a diffeomorphism $\phi \colon M \to M$ with $\phi(\iota^\prime(s_1)) = q$ and $\phi(p) = p$. Now set $\iota := \phi \circ \iota^\prime$.
\end{proof}

The above corollary should also be true if $M$ is allowed to have boundary, but the proof fails, since Corollary \ref{constr_diffeo:cor:transitivity_of_diffeo_groups} is clearly false if $M$ has non-empty boundary. An interesting consequence of Corollary \ref{constr_diffeo:cor:embeddings} is the existence of \textbf{embedded paths}: For any connected manifold $M$ (without boundary) and any $p,q \in M$ there exists an embedding $\gamma \colon [0,1] \to M$ with $\gamma(0) = p, \gamma(1) = q$.

By the Whitney embedding theorem any $n$-dimensional manifold can be embedded into $\R^{2n+1}$, but there might be different non-isotopic embeddings. Think for example of inequivalent knots in $\R^3$. The following theorem however says, that upon further increasing the dimension by one, there is only one embedding $M \hookrightarrow \R^{2n+2}$ up to isotopy (at least for $M$ closed).

\begin{theorem}
    \label{constr_diffeo:thm:equivalence_of_embeddings}
    Let $M$ be a closed $n$-dimensional manifold. Then any two embeddings $M \hookrightarrow \R^{2n+k}$ for $k \geq 2$ are isotopic.
\end{theorem}

\begin{proof}
    Let $N:= 2n + k$ and let $f,g \colon M \hookrightarrow \R^N$ be two embeddings. The idea of the proof is to construct preliminary isotopies $f \simeq f^\prime$ and $g \simeq g^\prime$, such that the straight line homotopy
    \[
    h_t := (1-t) f^\prime + t g^\prime
    \]
    is an isotopy. Let us observe what is necessary for $h_t$ to be an isotopy. First of all injectivity of $h_t$ for each time $t \in [0,1]$ is equivalent to (note that for $t = 0,1$ there is nothing to show)
    \[
    (1-t)(f^\prime(x) - f^\prime(y)) \neq t(g^\prime(y) - g^\prime(x))
    \]
    for all $t \in (0,1)$ and $x,y \in M$ with $x \neq y$. If we denote by $\Delta(X) \subseteq X \times X$ the diagonal of a manifold $X$ this in turn is equivalent to 
    \[
    ([f^\prime(x) - f^\prime(y)], [g^\prime(x) - g^\prime(y)]) \notin \Delta(\R P^{N-1}) \subseteq \R P^{N-1} \times \R P^{N-1}.
    \]
    But this precisely means that the map
    \begin{align*}
    M \times M - \Delta(M) & \to \R P^{N-1} \times \R P^{N-1} \\
    (x,y) &\mapsto ([f^\prime(x) - f^\prime(y)], [g^\prime(x) - g^\prime(y)])
    \end{align*}
    is transverse to the diagonal $\Delta(\R P^{N-1})$, since 
    \begin{align*}
        \mathrm{dim}(\Delta(\R P^{N-1})) + \mathrm{dim}(M \times M - \Delta(M)) & = 4n + k -1 \\ 
        &< 4n + 2(k-1) = \mathrm{dim}(\R P^{N-1} \times \R P^{N-1}).
    \end{align*}
    Similiarly injectivity of the differential of $h_t$ is equivalent to 
    \[
    (1-t) d_xf^\prime (v) \neq -t d_xg^\prime (v)
    \]
    for all $t \in (0,1)$ and $(x,v) \in TM - M$, where we embedd $M \subseteq TM$ as the zero section. Again we can formulate this as a transversality condition, namely that the map 
   \begin{align*}
    TM -M & \to \R P^{N-1} \times \R P^{N-1} \\
    (x,v) &\mapsto ([d_xf^\prime (v)], [d_xg^\prime (v)])
    \end{align*}
    is transversal to the diagonal $\Delta(\R P^{N-1})$. In order to find such maps $f^\prime$ and $g^\prime$ we will be using the parametric transversality theorem. More precisely we will show that for almost all matrices $A,B \in \mathrm{GL}(N)$ the maps 
    \[
    f^\prime := A \circ f, \; g^\prime := A \circ g
    \]
    satisfy the desired transversality conditions. Note that for $A, B \in \mathrm{GL}^+(N)$ we have 
    \[
    f \simeq A \circ f, \; g \simeq A \circ g,
    \]
    since $\mathrm{GL}^+(N)$ is the path connected component of $\mathrm{GL}(N)$ containing the identity. Now consider the maps 
    \begin{align*}
        \Phi \colon (M \times M - \Delta(M)) \times \mathrm{GL}(N) \times \mathrm{GL}(N) &\to (\R^N - 0) \times (\R^N - 0)\\
        (x,y,A,B) &\mapsto (A (f(x)- f(y)),B(g(x) - g(y)))
    \end{align*}
    and 
    \begin{align*}
        \Psi \colon (TM - M) \times \mathrm{GL}(N) \times \mathrm{GL}(N) &\to (\R^N - 0) \times (\R^N - 0)\\
        (x,v,A,B) &\mapsto (A \, d_xf(v),B\, d_xg(v)).
    \end{align*}
    To compute the derivatives let us write $L_w$ for the map $L_w \colon \R^{N \times N} \to \R^N, \; S \mapsto Sw$ for $w \in \R^N$, which is clearly surjective for $w \neq 0$. Then 
    \[
    d_{(x,y,A,B)} \Phi = \begin{pmatrix}
        \ast & L_{f(x)-f(y)} & 0 \\
        \ast & 0 & L_{g(x)-g(y)}
    \end{pmatrix}
    \]
    as a map $$T_{(x,y)} (M \times M - \Delta(M)) \oplus \R^{N \times N} \oplus \R^{N \times N} \to \R^N \oplus \R^N,$$ where we denote by $\ast$ the entries irrelevant for our discussion. Since $f(x)-f(y), g(x)-g(y) \neq 0$ it follows that this map is surjective and thus $\Phi$ is a submersion. In particular, post composing with the projection
    \[
    \pi \colon (\R^N - 0) \times (\R^N - 0) \to \R P^{N-1} \times \R P^{N-1}
    \]
    shows that $\pi \circ \Phi$ is a submersion. Essentially the same calculation also shows that $\pi \circ \Psi$ is a submersion. Now the theorem follows from the parametric transversality theorem.
\end{proof}

% Following \cite{hirsch2012differential} we call an embedding $\iota \colon M \hookrightarrow N$ of manifolds with boundary a \textbf{neat embedding} if $\iota(\partial M) = \iota(M) \cap \partial N$. 
As a corollary of the above theorem and the isotopy extension theorem we can show:

\begin{corollary}
    Let $W$ be an $n$-dimensional manifold with compact boundary. Then any embedding $\partial W \hookrightarrow \R^{2n+1}$ can be extended to an embedding $W \hookrightarrow \R^{2n+1}$.
\end{corollary}

\begin{proof}
    By the Whitney embedding theorem we can choose an embedding $\phi \colon W \hookrightarrow \R^{2n+1}$.
    %  Moreover let $f \colon W \to [0,\infty)$ be a smooth function which vanishes on the boundary $\partial W$ and satisfies $f > 0$ on the interior $\mathrm{int}(W)$. Then 
    % \[
    % \psi \colon W \hookrightarrow \mathbb{H}^{2n+2}, x \mapsto (\phi(x),f(x))
    % \]
    % is a neat embedding of $W$ and thus in particular 
    Let $f := \phi|_{\partial W}$ be the restriction to the boundary and let $g \colon \partial W \hookrightarrow \R^{2n+1}$ be any other embedding. By Theorem \ref{constr_diffeo:thm:equivalence_of_embeddings} there exists an isotopy from the inclusion $f(\partial W) \hookrightarrow \R^{2n+1}$ to the embedding 
    \[
    g \circ f^{-1} \colon f(\partial W) \hookrightarrow \R^{2n+1}
    \]
    and by the isotopy extension Theorem (Theorem \ref{constr_diffeo:thm:isotopy_extension_theorem}) we can extend this isotopy to an isotopy $h_t \colon \R^{2n+1} \to \R^{2n+1}$. Then $\psi := h_1 \circ \phi \colon W \hookrightarrow \R^{2n+1}$ is an extension of $g$.
\end{proof}
