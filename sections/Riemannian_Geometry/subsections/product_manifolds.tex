\subsection{Product Manifolds}

In the following let $(M_1,g_1)$ and $(M_2,g_2)$ be Riemannian manifolds. Then we can consider the product manifold 
\[
M := M_1 \times M_2
\]
equipped with the product metric $g:= g_1 \times g_2$ defined by 
\begin{equation}
    \label{riem_mnf:eq:product_metric}
g(X_1+X_2,Y_1+Y_2) = g_1(X_1,Y_1) + g_2(X_2,Y_2),
\end{equation}
where $X_i,Y_i \in \mathscr{X}(M_i)$. In this section we compute the Levi-Civita connection, curvature tensors, covariant derivative along curves and geodesics of $(M,g)$ and see how they relate to the corresponding objects on $(M_1,g_1)$ and $(M_2,g_2)$.

Let us establish some notation used in the discussion that follows. For a vector field $Z \in \mathscr{X}(M)$ we write $\tilde{Z} \in \mathscr{X}(M)$ for the horizontal lift of $Z$, that ist 
\[
\tilde{Z}|_{(p_1,p_2)} := Z|_{p_1}
\]
for $(p_1,p_2) \in M$. Moreover for a smooth function $f \in C^\infty(M_1)$ we define 
\[
\tilde{f}:= f \circ \pi_1 \in C^\infty(M),
\]
where $\pi_1 \colon M \to M_1$ denotes the projection. The same notation is used for vector fields and smooth functions on $M_2$. So to be precise, equation (\ref{riem_mnf:eq:product_metric}) should read 
\[
g(\tilde{X}_1+\tilde{X}_2,\tilde{Y}_1+\tilde{Y}_2) = \reallywidetilde{\mathit{g_1(X_1,Y_1)}}{7} + \reallywidetilde{\mathit{g_2(X_2,Y_2)}}{7}.
\]
For the sake of notational simplicity we will often suppress these subtleties, if there is no risk of confusion and only write $X_1 + X_2$ instead of $\tilde{X}_1+\tilde{X}_2$ or $f$ instead of $\tilde{f}$. It should always be clear from the context what is meant. Let us moreover denote by $\nabla^1,D_t^1,\mathrm{Rm}_1,\mathrm{Ric}_1,\mathrm{scal}_1$ the Levi-Civita connection, covariant derivative along curves, Riemannian curvature tensor, Ricci tensor and scalar curvature of $(M_1,g_1)$ respectively. Similiarly we denote the same objects on $(M_2,g_2)$ by $\nabla^2,D_t^2,\mathrm{Rm}_2,\mathrm{Ric}_2,\mathrm{scal}_2$ and on $(M,g)$ simply by $\nabla,D_t,\mathrm{Rm},\mathrm{Ric},\mathrm{scal}$. Finally we define $m_1 := \mathrm{dim}(M_1)$ and $m_2 := \mathrm{dim}(M_2)$.

Of course our first step will be to compute $\nabla$ in terms of $\nabla^1$ and $\nabla^2$. For this we begin with a more general statement on constructing a canonical connection on product vector bundles, where we use the same notational convention for sections in the product bundle as for vector fields introduced above.


% In the discussion that follows we use the notation $X_1+X_2$ and $(X_1,X_2)$ interchangebly. We may also occasionally write 
% \[
% \tilde{X}_1 := (X_1,0) \in \mathscr{X}(M)
% \]
% for the horizontal lift of $X_1 \in \mathscr{X}(M_1)$, but usually it should be clear from the context if we consider $X_1$ as a vector field on $M_1$ or its horizontal lift as a vector field on $M$, in which case we refrain from differentiating these two notationally. 

\begin{proposition}
    \label{riem_mnf:prop:product_connections}
    Let $E_1 \to M_1$ and $E_2 \to M_2$ be smooth vector bundles and let $\nabla^1, \nabla^2$ be connections on $E_1,E_2$ respectively. Then there exists a unique connection $\nabla$ on the product bundle
    \[
    E_1\times E_2 \to M_1\times M_2,
    \]
    such that for sections $s \in \Gamma(E_1\times E_2)$ of the form $s = s_1 + s_2$, where $s_i \in \Gamma(E_i)$, and vector fields $X \in \mathscr{X}(M_1\times M_2)$ of the form $X=X_1 + X_2$, where $X_i \in \mathscr{X}(M_i)$, the following holds:
    \[
    \nabla_X s = \nabla^1_{X_1}s_1 + \nabla^2_{X_2}s_2.
    \]
    We call $\nabla$ the product connection of $\nabla^1$ and $\nabla^2$ and write $\nabla = \nabla^1 \times \nabla^2$.
\end{proposition}

\begin{proof}
    We first show uniqueness. Let $X \in \mathscr{X}(M_1\times M_2)$ be an arbitary vector field and $s \in \Gamma(E_1\times E_2)$ an arbitary section. For $(p_1,p_2) \in M_1 \times M_2$ we can find open neighbourhoods $U_i \subseteq M_i$ of $p_i$, such that there exist local frames 
    \[
    \sigma_1^i,\ldots,\sigma_{n_i}^i \in \Gamma(E_i|_{U_i}),
    \]
    where $n_i$ is the rank of $E_i$ and there exist local frames 
    \[
    E_1^i,\ldots,E_{m_i}^i \in \Gamma(TM_i|_{U_i}),
    \]
    On $U:= U_1 \times U_2$ we can write
    \[
    s = s_1^{j_1}\sigma_{j_1}^1 + s_2^{j_2}\sigma_{j_2}^2
    \]
    and
    \[
    X = X_1^{k_1}E_{k_1}^1 + X_2^{k_2}E_{k_2}^2
    \]
    for smooth functions $s_1^{j_1},s_2^{j_2},X_1^{k_1},X_2^{k_2} \in C^\infty(U)$. Then
    \begin{align}
        \nabla_X s &= X_1^{k_1}\left(\nabla_{E_{k_1}^1}(s_1^{j_1}\sigma_{j_1}^1) + \nabla_{E_{k_1}^1}(s_2^{j_2}\sigma_{j_2}^2)\right) + X_2^{k_2}\left(\nabla_{E_{k_2}^2}(s_1^{j_1}\sigma_{j_1}^1) + \nabla_{E_{k_2}^2}(s_2^{j_2}\sigma_{j_2}^2)\right) \notag \\
        &= X_1^{k_1}\left((E_{k_1}^1 \cdot s_1^{j_1})\sigma_{j_1}^1 + s_1^{j_1}\nabla^1_{E_{k_1}^1}\sigma_{j_1}^1 + (E_{k_1}^1 \cdot s_2^{j_2})\sigma_{j_2}^2\right) \notag\\
        &\phantom{=} + X_2^{k_2}\left((E_{k_2}^2 \cdot s_1^{j_1})\sigma_{j_1}^1 + (E_{k_2}^2 \cdot s_2^{j_2})\sigma_{j_2}^2 + s_2^{j_2}\nabla^2_{E_{k_2}^2}\sigma_{j_2}^2 \right) \label{riem_mnf:eq:product_connection}
    \end{align}
    Since the last expression does not depend on $\nabla$ this shows uniqueness. To show existence define a connection $\nabla^U$ on $U$ by formula (\ref{riem_mnf:eq:product_connection}). It is easy to see that this does in fact define a connection on $E_1 \times E_2|_U$, which is compatible with the restricted connections $\nabla^1|_{U_1}$ and $\nabla^2|_{U_2}$ in the sense of the statement of the proposition. So by the uniqueness result this connection does not depend on the chosen frames, thus we get a well-defined connection $\nabla$ on $E_1\times E_2$ by 
    \[
    (\nabla_Xs)|_U := \nabla^U_{X|_U}s|_U.
    \]
\end{proof}

\begin{proposition}
    \label{riem_mnf:prop:levi_civita}
    The Levi-Civita connection $\nabla$ on $(M,g)$ is given by
    \[
    \nabla = \nabla^1 \times \nabla^2.
    \]
\end{proposition}

\begin{proof}
    To ensure clarity, we will be more precise with the notation in this proof, as explained at the beginning of the section. Let $X,Y \in \mathscr{X}(M)$ be of the form $X=\tilde{X}_1+\tilde{X}_2, Y=\tilde{Y}_1+\tilde{Y}_2$, where $X_i,Y_i \in \mathscr{X}(M_i)$. By Proposition \ref{riem_mnf:prop:product_connections} it suffices to show that the Levi Civita connection satisfies
    \[
    \nabla_XY = \reallywidetilde{\nabla^1_{X_1}Y_1}{5} + \reallywidetilde{\nabla^2_{X_2}Y_2}{5}.
    \]
    We would like to use the Koszul formula to compute $\nabla_XY$ and begin by simplifying some of the terms involved in this formula. For $Z \in \mathscr{X}(M_1), W \in \mathscr{X}(M_2)$ we claim that 
    \[
    [\tilde{Z},\tilde{W}] = 0.
    \] 
    To see this choose coordinates $x^1,\ldots,x^{m_1}$ of $M_1$ and $y^1,\ldots,y^{m_2}$ of $M_2$. Write 
    \[
    Z= f^i \frac{\partial}{\partial x^i}, \; W= g^j \frac{\partial}{\partial y^j}
    \] 
    for smooth functions $f^i$ and $g^j$ defined on the chosen coordinate neighbourhoods of $M_1$ and $M_2$ respectively. Then $x^1,\ldots,x^{m_1},y^1,\ldots,y^{m_2}$ define coordinates of $M$ and 
    \[
    \tilde{Z}= \tilde{f}^i \frac{\partial}{\partial x^i}, \; \tilde{W}= \tilde{g}^j \frac{\partial}{\partial y^j}.
    \]
    We now compute 
    \begin{align*}
        [\tilde{Z},\tilde{W}] &= \tilde{f}^i\frac{\partial \tilde{g}^j}{\partial x^i}\frac{\partial}{\partial y^j} + \tilde{g}^j\left[\tilde{f}^i \frac{\partial}{\partial x^i},\frac{\partial}{\partial y^j}\right] \\
        &= \tilde{f}^i\frac{\partial \tilde{g}^j}{\partial x^i}\frac{\partial}{\partial y^j} - \tilde{g}^j\frac{\partial \tilde{f}^i}{\partial y^j}\frac{\partial}{\partial x^i} - \tilde{g}^j\tilde{f}^i \left[ \frac{\partial}{\partial y^j},\frac{\partial}{\partial x^i} \right].
    \end{align*}
    Since $\tilde{g}^j$ does not depend on $x^i$ we get 
    $$\frac{\partial \tilde{g}^j}{\partial x^i} =0.$$ 
    Similiarly 
    \[
    \frac{\partial \tilde{f}^i}{\partial y^j} =0,
    \]
    such that $[\tilde{Z},\tilde{W}] = 0$ follows. For $Z$ as above the Koszul formula now reads 
    \begin{align}
        g(\nabla_XY,\tilde{Z}) &= \frac{1}{2}\Big(Xg(Y,\tilde{Z}) + Yg(\tilde{Z},X) - \tilde{Z}g(X,Y) \notag \\
        &\phantom{=} - g(Y,[X,\tilde{Z}]) - g(\tilde{Z},[Y,X]) + g(X,[\tilde{Z},Y])\Big) \notag \\
        &= \frac{1}{2}\Big(\tilde{X}_1g(\tilde{Y}_1,\tilde{Z}) + \tilde{Y}_1g(\tilde{Z},\tilde{X}_1) - \tilde{Z}g(\tilde{X}_1,\tilde{Y}_1) \notag \\
        &\phantom{=} - g(\tilde{Y}_1,[\tilde{X}_1,\tilde{Z}]) - g(\tilde{Z},[\tilde{Y}_1,\tilde{X}_1]) + g(\tilde{X}_1,[\tilde{Z},\tilde{Y}_1])\Big) \label{riem_mnf:eq:koszul1}
    \end{align}
    Besides the claim about the Lie bracket shown above we have further used, that $g(Y,\tilde{Z}) = g(\tilde{Y}_1,\tilde{Z})$ is constant in the $M_2$-direction, such that $\tilde{X}_2\,g(\tilde{Y}_1,\tilde{Z}) = 0$, hence 
    \[
    Xg(Y,\tilde{Z}) = \tilde{X}_2\,g(\tilde{Y}_1,\tilde{Z}).
    \]
    An analogous statement holds for the second and third term. Moreover $[\tilde{X}_1,\tilde{Z}] = \reallywidetilde{[X,Z]}{4.5}$, such that $g(\tilde{Y}_2,[\tilde{X}_1,\tilde{Z}]) = 0$, hence 
    \[
    g(Y,[X,\tilde{Z}]) = g(\tilde{Y}_1,[\tilde{x}_1,\tilde{Z}]).
    \]
    Again analogous statements hold for the last two terms. Applying the Koszul formula to $\nabla^1_{X_1}Y_1$ we get  
    \begin{align}
        g_1(\nabla^1_{X_1}Y_1,Z) &= \frac{1}{2}\Big(X_1g_1(Y_1,Z) + Y_1g_1(Z,X_1) - Zg_1(X_1,Y_1) \notag \\
        &\phantom{=} - g_1(Y_1,[X_1,Z]) - g_1(Z,[Y_1,X_1]) + g_1(X_1,[Z,Y_1])\Big) \label{riem_mnf:eq:koszul2}
    \end{align}
    To compare this expression to (\ref{riem_mnf:eq:koszul1}) we note the following identities for $U,V \in \mathscr{X}(M_1)$ and $f \in C^\infty(M_1)$:
    \[
    g(\tilde{U},\tilde{V}) = \reallywidetilde{g_1(U,V)}{6}
    \]
    just by definition of $g$ and 
    \[
    \tilde{V}\tilde{f} = \widetilde{Vf}.
    \]
    So (\ref{riem_mnf:eq:koszul1}) and (\ref{riem_mnf:eq:koszul2}) now yield 
    \[
    g(\nabla_XY,\tilde{Z}) = \reallywidetilde{g_1(\nabla^1_{X_1}Y_1,Z)}{9} = g(\reallywidetilde{\nabla^1_{X_1}Y_1}{5},\tilde{Z}).
    \]
    Similiarly we get for $W \in \mathscr{X}(M_2)$
    \[
    g(\nabla_XY,\tilde{Z}) = g(\reallywidetilde{\nabla^2_{X_2}Y_2}{5},\tilde{W}).
    \]
    Together this implies
    \[
    g(\nabla_XY,\tilde{Z}+\tilde{W}) = g(\reallywidetilde{\nabla^1_{X_1}Y_1}{5} + \reallywidetilde{\nabla^2_{X_2}Y_2}{5},\tilde{Z}+\tilde{W}).
    \]
    From which the assertion follows.

\end{proof}

Knowing the Levi-Civita connection on $M$, it is now easy to compute the curvature, which behaves just as expected.

\begin{corollary}
    Let $X = X_1 + X_2,\, Y = Y_1 + Y_2, \,Z = Z_1 + Z_2,\, W = W_1 + W_2$ be smooth vector fields on $M$, where $X_i,Y_i,Z_i,W_i \in \mathscr{X_i}$. Then the following holds
    \begin{enumerate}[(i)]
        \item $\mathrm{Rm}(X,Y,Z,W) = \mathrm{Rm}_1(X_1,Y_1,Z_1,W_1) + \mathrm{Rm}_2(X_2,Y_2,Z_2,W_2)$
        \item $\mathrm{Ric}(X,Y) = \mathrm{Ric}_1(X_1,Y_1) + \mathrm{Ric}_2(X_2,Y_2)$
        \item $\mathrm{scal} = \mathrm{scal}_1 + \mathrm{scal}_2$
    \end{enumerate}
\end{corollary}

\begin{proof}
    By Proposition \ref{riem_mnf:prop:levi_civita}
    \[
    \nabla_X\nabla_YZ = \nabla_X(\nabla^1_{Y_1}Z_1 + \nabla^2_{Y_2}Z_2) = \nabla_{X_1}^1\nabla^1_{Y_1}Z_1 + \nabla_{X_2}^2\nabla^2_{Y_2}Z_2.
    \]
    An analogous equation holds for $\nabla_Y\nabla_XZ$. Moreover
    \[
    [X,Y] = [X_1,Y_1] + [X_2,Y_2]
    \]
    as we have seen in the proof of Proposition \ref{riem_mnf:prop:levi_civita}, such that 
    \[
    \nabla_{[X,Y]}Z = \nabla^1_{[X_1,Y_1]}Z_1 +  \nabla^2_{[X_2,Y_2]}Z_2.
    \]
    Then $(i)$ follows from the definition of Rm and the product metric $g$. For $(ii)$ write 
    \[
    \mathrm{Ric}(X,Y) = \sum_{i=1}^{m_1} \mathrm{Rm}(X,E_i^1,E_i^1,Y) + \sum_{j=1}^{m_2} \mathrm{Rm}(X,E_j^2,E_j^2,Y)
    \]
    for local orthonormal frames $E_1^1,\ldots,E_{m_1}^1$ of $TM_1$ and $E_2^2,\ldots,E_{m_2}^2$ of $TM_2$. By $(i)$
    \begin{align*}
        \mathrm{Rm}(X,E_i^1,E_i^1,Y) &= \mathrm{Rm}_1(X_1,E_i^1,E_i^1,Y_1) + \mathrm{Rm}_2(X_2,0,0,Y_2)\\
        &= \mathrm{Rm}_1(X_1,E_i^1,E_i^1,Y_1).
    \end{align*}
    An analogous statement holds for the second term, such that 
    \begin{align*}
        \mathrm{Ric}(X,Y) &= \sum_{i=1}^{m_1} \mathrm{Rm}_1(X_1,E_i^1,E_i^1,Y_1) + \sum_{j=1}^{m_2} \mathrm{Rm}_2(X_2,E_j^2,E_j^2,Y_2) \\
        &= \mathrm{Ric}_1(X_1,Y_1) + \mathrm{Ric}_2(X_2,Y_2).
    \end{align*}
    Finally $(iii)$ follows from $(ii)$ in the same way.
\end{proof}

Finally we compute the covariant derivative along curves and geodesics of $M$, which again behave in exactly the way one would expect. 

\begin{corollary}
    \label{riem_mnf:cor:covariant_derivative_curves}
    Let $\gamma = (\gamma_1,\gamma_2) \colon I \to M$ be a smooth curve and $V \colon I \to \gamma^\ast TM \cong \gamma_1^\ast TM_1 \oplus \gamma_2^\ast TM_2$ a smooth vector field along $\gamma$. Write $V= V_1 + V_2$ for $V_i \colon I \to \gamma_i^\ast TM$. Then 
    \[
    D_tV = D_t^1V_1 + D_t^2V_2.
    \]
\end{corollary}

\begin{proof}
    As in the proof of Proposition \ref{riem_mnf:prop:levi_civita}, we will take some care with the notation in this proof. To show the statement we compute both sides locally. To this end let $E_1^1,\ldots,E_{m_1}^1$ be a local frame of $TM_1$ and $E_2^2,\ldots,E_{m_2}^2$ a local frame of $TM_2$. Write 
    \[
    V_1(t) = V_1^i(t) E_i^1|_{\gamma_1(t)}
    \]
    and 
    \[
    V_2(t) = V_2^j(t) E_j^2|_{\gamma_2(t)}.
    \]
    Then 
    \[
    V(t) = V_1^i(t) \tilde{E}_i^1|_{\gamma(t)} + V_2^j(t) \tilde{E}_j^2|_{\gamma(t)}
    \]
    and hence 
    \begin{align*}
        D_tV & = \dot{V}_1^i(t)\tilde{E}_i^1|_{\gamma(t)} + V_1^i(t)\nabla_{\dot{\gamma}(t)}\tilde{E}_i^1 \\[.3em]
        &\phantom{=}+ \dot{V}_2^j(t)\tilde{E}_j^2|_{\gamma(t)} + V_2^j(t)\nabla_{\dot{\gamma}(t)}\tilde{E}_j^2.
    \end{align*}
Moreover 
\[
D_t^1V_1 = \dot{V}_1^i(t)E_i^1|_{\gamma_1(t)} + V_1^i(t)\nabla^1_{\dot{\gamma}_1(t)}E_i^1
\]
and similiarly for $D_t^2V_2$. By Proposition \ref{riem_mnf:prop:levi_civita}
\[
\nabla_{\dot{\gamma}(t)}\tilde{E}_i^1 = \reallywidetilde{\nabla^1_{\dot{\gamma}_1(t)}E_i^1}{6} \quad \text{ and } \quad \nabla_{\dot{\gamma}(t)}\tilde{E}_j^2 = \reallywidetilde{\nabla^2_{\dot{\gamma}_2(t)}E_j^2}{6}
\]
and thus 
\[
D_tV = \reallywidetilde{D_t^1V_1}{4} + \reallywidetilde{D_t^2V_2}{4}.
\]
\end{proof}

\begin{corollary}
    Let $\gamma = (\gamma_1,\gamma_2) \colon I \to M$ be a smooth curve. Then $\gamma$ is a geodesic if and only if $\gamma_1$ and $\gamma_2$ are geodesics.
\end{corollary}
\begin{proof}
   This now follows immediately form the formula from Corollary \ref{riem_mnf:cor:covariant_derivative_curves}.
\end{proof}
