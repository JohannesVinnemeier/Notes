\subsection{Induced connections}
Given any vector bundle there are many ways to construct new ones from it. In this section we show how a connection on the new bundle is induced by a connection on the bundle we started with.

In the following proposition we shall use the same notational convention for sections in the product bundle as for vector fields
introduced at the beginning of the last section.

\begin{proposition}
    \label{riem_mnf:prop:product_connections}
    Let $E_1 \to M_1$ and $E_2 \to M_2$ be smooth vector bundles and let $\nabla^1, \nabla^2$ be connections on $E_1,E_2$ respectively. Then there exists a unique connection $\nabla$ on the product bundle
    \[
    E_1\times E_2 \to M_1\times M_2,
    \]
    such that for sections $s \in \Gamma(E_1\times E_2)$ of the form $s = s_1 + s_2$, where $s_i \in \Gamma(E_i)$, and vector fields $X \in \mathscr{X}(M_1\times M_2)$ of the form $X=X_1 + X_2$, where $X_i \in \mathscr{X}(M_i)$, the following holds:
    \[
    \nabla_X s = \nabla^1_{X_1}s_1 + \nabla^2_{X_2}s_2.
    \]
    We call $\nabla$ the product connection of $\nabla^1$ and $\nabla^2$ and write $\nabla = \nabla^1 \times \nabla^2$.
\end{proposition}

\begin{proof}
    We first show uniqueness. Let $X \in \mathscr{X}(M_1\times M_2)$ be an arbitary vector field and $s \in \Gamma(E_1\times E_2)$ an arbitary section. For $(p_1,p_2) \in M_1 \times M_2$ we can find open neighbourhoods $U_i \subseteq M_i$ of $p_i$, such that there exist local frames 
    \[
    \sigma_1^i,\ldots,\sigma_{n_i}^i \in \Gamma(E_i|_{U_i}),
    \]
    where $n_i$ is the rank of $E_i$ and there exist local frames 
    \[
    E_1^i,\ldots,E_{m_i}^i \in \Gamma(TM_i|_{U_i}),
    \]
    On $U:= U_1 \times U_2$ we can write
    \[
    s = s_1^{j_1}\sigma_{j_1}^1 + s_2^{j_2}\sigma_{j_2}^2
    \]
    and
    \[
    X = X_1^{k_1}E_{k_1}^1 + X_2^{k_2}E_{k_2}^2
    \]
    for smooth functions $s_1^{j_1},s_2^{j_2},X_1^{k_1},X_2^{k_2} \in C^\infty(U)$. Then
    \begin{align}
        \nabla_X s &= X_1^{k_1}\left(\nabla_{E_{k_1}^1}(s_1^{j_1}\sigma_{j_1}^1) + \nabla_{E_{k_1}^1}(s_2^{j_2}\sigma_{j_2}^2)\right) + X_2^{k_2}\left(\nabla_{E_{k_2}^2}(s_1^{j_1}\sigma_{j_1}^1) + \nabla_{E_{k_2}^2}(s_2^{j_2}\sigma_{j_2}^2)\right) \notag \\
        &= X_1^{k_1}\left((E_{k_1}^1 \cdot s_1^{j_1})\sigma_{j_1}^1 + s_1^{j_1}\nabla^1_{E_{k_1}^1}\sigma_{j_1}^1 + (E_{k_1}^1 \cdot s_2^{j_2})\sigma_{j_2}^2\right) \notag\\
        &\phantom{=} + X_2^{k_2}\left((E_{k_2}^2 \cdot s_1^{j_1})\sigma_{j_1}^1 + (E_{k_2}^2 \cdot s_2^{j_2})\sigma_{j_2}^2 + s_2^{j_2}\nabla^2_{E_{k_2}^2}\sigma_{j_2}^2 \right) \label{riem_mnf:eq:product_connection}
    \end{align}
    Since the last expression does not depend on $\nabla$ this shows uniqueness. To show existence define a connection $\nabla^U$ on $U$ by formula (\ref{riem_mnf:eq:product_connection}). It is easy to see that this does in fact define a connection on $E_1 \times E_2|_U$, which is compatible with the restricted connections $\nabla^1|_{U_1}$ and $\nabla^2|_{U_2}$ in the sense of the statement of the proposition. So by the uniqueness result this connection does not depend on the chosen frames, thus we get a well-defined connection $\nabla$ on $E_1\times E_2$ by 
    \[
    (\nabla_Xs)|_U := \nabla^U_{X|_U}s|_U.
    \]
\end{proof}

The next propostion is very similiar and concerned with the induced connection on a tensor bundle.

\begin{proposition}
    Let $E_1 \to M$ and $E_2 \to M$ be smooth vector bundles and let $\nabla^1, \nabla^2$ be connections on $E_1,E_2$ respectively. Then there exists a unique connection $\nabla$ on the tensor product bundle
    \[
    E_1\otimes E_2 \to M,
    \]
    such that for any vector field $X \in \mathscr{X}(M)$ and sections $s_1 \in \Gamma(E_1)$ and $s_2 \in \Gamma(E_2)$ the following holds:
    \[
    \nabla_X (s_1 \otimes s_2) = \nabla^1_{X}s_1 \otimes s_2 + s_1 \otimes \nabla^2_{X}s_2.
    \]
\end{proposition}
\begin{proof}
    The proof is analogous to the proof of Proposition \ref{riem_mnf:prop:product_connections}, so we only summarize the idea brielfy. One notes that any section $s \in \Gamma(E_1 \otimes E_2)$ can be written locally as 
    \[
    s = \sum_{i,j} s_1^i \otimes s_2^j,
    \]
    where $s_1^i$ and $s_2^j$ are local sections of $E_1$ and $E_2$ respectively. This then shows uniqueness and moreover gives us a way to define $\nabla$ locally, which can be pasted to together to define $\nabla$ globally just as in Proposition \ref{riem_mnf:prop:product_connections}.
\end{proof}

Finally we discuss the induced connection on the pullback bundle $f^\ast E$ of some vector bundle $E \to M$ under a smooth map $f \colon N \to M$. For this we introduce the following notation: given a section $s$ of $E$ we can define a section $f^\ast s$ of $f^\ast E$ by $(f^\ast s)(p) := s(f(p))$.

To conclude this section we present two useful formulas for induced connections.

\begin{lemma}
    Let $M$ be a Riemannian manifold, $\omega,\eta \in \Omega^\ast(M)$ differential forms and $X \in \mathscr{X}(M)$ a vector field. Then 
    \[
    \nabla_X(\omega \wedge \eta) = \nabla_X \omega \wedge \eta + \omega \wedge \nabla_X\eta,
    \]
    where $\nabla_X$ denotes the connection on the bundle $\Lambda^\ast T^\ast M$ induced by the Levi-Civita connection on $M$.
\end{lemma}

\begin{proof}
    By definition of the wedge product it suffices to show that $\nabla_X$ commutes with the antisymmetrization map 
    \[
    \mathrm{Alt} \colon (T^\ast M)^{\otimes k} \to (T^\ast M)^{\otimes k}.
    \]
    To see this let $\alpha_1,\ldots,\alpha_k \in \Omega^1(M)$. Then 
    \begin{align*}
        \nabla_X (\mathrm{Alt}(\alpha_1\otimes\ldots\otimes\alpha_k)) &= \nabla_X\left(\frac{1}{k!}\sum_{\sigma} \mathrm{sgn}(\sigma) \alpha_{\sigma(1)} \otimes \ldots \otimes \alpha_{\sigma(k)}\right) \\
        &= \frac{1}{k!}\sum_{\sigma}\sum_{i}\mathrm{sgn}(\sigma) \alpha_{\sigma(1)} \otimes \ldots \otimes \nabla_X\alpha_{\sigma(i)}\otimes \ldots \otimes \alpha_{\sigma(k)}
    \end{align*}
    and 
    \begin{align*}
        \mathrm{Alt}(\nabla_X (\alpha_1\otimes\ldots\otimes\alpha_k)) &= \mathrm{Alt}\left(\sum_i \alpha_1 \otimes \ldots \otimes \nabla_X\alpha_i \otimes \ldots \otimes \alpha_k\right) \\
        &= \sum_{i}\sum_{\sigma}\frac{1}{k!}\mathrm{sgn}(\sigma) \alpha_{\sigma(1)} \otimes \ldots \otimes \nabla_X\alpha_{\sigma(i)}\otimes \ldots \otimes \alpha_{\sigma(k)}.
    \end{align*}
    Since locally any section $\omega \in \Gamma((T^\ast M)^{\otimes k})$ can be written as a sum of tensor products of $1$-forms the statement follows.
\end{proof}

\begin{lemma}
    Let $E \to M$ be a smooth vector bundle equipped with a metric and compatible connection. Then for any $X \in \mathscr{X}(M)$ the following diagram commutes 
    \[\begin{tikzcd}
	{\Gamma(E)} & {\Gamma(E)} \\
	{\Gamma(E^\ast)} & {\Gamma(E^\ast)}
	\arrow["{\nabla_X}", from=1-1, to=1-2]
	\arrow["\cong", from=1-1, to=2-1]
	\arrow["\cong", from=1-2, to=2-2]
	\arrow["{\nabla_X}", from=2-1, to=2-2]
\end{tikzcd}\]
where the vertical isomorphisms are given by dualizing and the connection on $E^\ast$ is the unique connection defined by
\[
X\cdot (\alpha(s)) = (\nabla_X\alpha)(s) + \alpha(\nabla_Xs)
\] 
for $s \in \Gamma(E), \alpha \in \Gamma(E^\ast)$.
\end{lemma}
\begin{proof}
    This follows immediately from the definition of the connection on $E^\ast$ and the compatibility of the metric.
\end{proof}
