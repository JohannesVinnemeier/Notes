\section{Solutions to selected exercises}


\begin{exercise}[Exercise 10-10 from \cite{lee2011smooth}]
Let $M$ be a manifold without boundary of dimension $n$ and $E \to M$ a rank $k$ vector bundle over $M$. Then there exists a section $\sigma \colon M \to E$ with the following properties:
\begin{enumerate}
    \item If $k > n$, then $\sigma$ is nowhere vanishing
    \item If $k \leq n$, then the zeros of $\sigma$ form a closed codimension $k$ submanifold of $M$
\end{enumerate}
In particular, there exists a smooth vector field $X$ on $M$, where the zeros of $X$ are a discrete closed subset. Hence, every compact manifold with or without boundary admits a smooth vector field with only finitely many zeros. 
\end{exercise}

\begin{proof}
    Consider the zero section $\xi \colon M \to E$ and let $M^\prime := \mathrm{im}(\xi)$. Then $M^\prime$ is a codimension $0$ submanifold (without boundary) of $E$. By the transversality homotopy theorem (Theorem \ref{thm:transversality_homotopy}) any section $M \to E$ is homotopic to a section $\sigma \colon M \to E$ that is transversal to $M^\prime$. If $k > n$ this is equivalent to 
    \[
    \sigma(M) \cap M^\prime = \emptyset,
    \]
    which in turn is equivalent to $\sigma(p) \neq 0$ for all $p \in M$. If $k \leq M$, then 
    \[\{p \in M \mid \sigma(p) = 0\} = \sigma^{-1}(M^\prime) \]
    and $\sigma^{-1}(M^\prime)$ is a closed codimension $k$ submanifold of $M$. This shows the first two statements. Applying these results to $E := TM$, we see that the zeros of $\sigma$ are a closed codimension $0$ submanifold, thus discrete. If $M$ is compact this is equivalent to $\{p \in M \mid \sigma(p) = 0\}$ being finite. If $M$ has non-empty boundary, we can construct a vector field with finitely many zeros on the double and restrict it to $M$.  
\end{proof}


\begin{exercise}[Own exercise]
    \label{ex:pullback_of_moebius_bundle}
    Let $M$ be the Moebius line bundle. Consider the map $f \colon S^1 \to S^1, \, z \mapsto z^2$. Then the pullback bundle $E := f^*M$ is trivial. 
\end{exercise}
Intuitively, we can see this as follows: The fiber over $e^{\pi i}$ of $E$ is just the fiber over $e^{2 \pi i}$ of $M$, which has already made a half-twist, i.e. the fiber of $E$ over $1$ can be imagined as a straight line and then begins twisting upon increasing the angle. When the angle $\theta = \pi$ is reached a half twist has been performed. But upon further increasing the angle, another half twist is performed as we "walk" around the moebius bundle a second time. So $E$ is basically a double twisted moebius bundle, which is trivial.

\begin{proof}{Proof of Exercise \ref{ex:pullback_of_moebius_bundle}}
    We define $M := \R \times \R/_\sim $, where the equivalence relation is given by 
    \[
    (x,t) \sim (x + k, (-1)^kt)
    \]
    and the projection $p \colon M \to S^1$ is defined by $p([(x,t)]) := e^{2 \pi i x}$.
    By identifiying $S^1$ with $[0,1]/_{0 \sim 1}$ we can write $f$ as
    \[
    f([\theta]) = e^{2\pi i \theta},
    \]
    such that $E$ is a line bundle over $[0,1]/_\sim$. To see that $E$ is trivial we define a global frame:
    \[
    \sigma \colon [0,1]/_\sim \to E, \; [\theta] \mapsto ([\theta],[(2\theta,1)])
    \]
    Note that $[(0,1)] = [(2,1)]$ in $M$, such that $\sigma$ is a well-defined section. Furthermore, it is easily seen that this section is nowhere vanishing. To see that $\sigma$ is smooth we could write $\sigma$ in the local trivialisations of $E$. Alternatively we can argue as follows: Consider $$s \colon [0,1] \to M, \; \theta \mapsto [(2\theta,1)]$$ and let $\pi \colon [0,1] \to [0,1]/_\sim$ be the projection map. Then $s = \sigma \circ \pi$. Clearly $s$ is smooth and it is easily verified that $\pi$ is a local diffeomorphism, such that smoothness of $\sigma$ follows. This completes the proof. 
\end{proof}

\begin{exercise}[Exercise 10-14 from \cite{lee2011smooth}]
    Let $M$ be a smooth manifold and $S \subseteq M$ a submanifold. Then $TS$ is a subbundle of $TM|_S$.
\end{exercise}
\begin{proof}
    Let $\sigma_1,\ldots,\sigma_k$ be a local frame for $TS$. Then we can also consider the $\sigma_i$ as local sections of $TM|_S$ via the inclusion $TS \hookrightarrow TM|_S$. If we can show that these are still smooth sections we are done as this shows that $TS$ (as a set) is a subbundle of $TM|_S$ an the induced smoothness structure is the same as the usual one on $TS$ (see the construction of the smoothness structure on the subbundle in Lemma 10.32 in \cite{lee2011smooth} to see that this indeed yields the same structures). But smoothness of $\sigma_i$ as local sections of $TM|_S$ follows directly from the fact that the inclusion $TS \hookrightarrow TM|_S$ is just the restriction of the differential of the inclusion $\iota \colon S \hookrightarrow M$ to the submanifold $TM|_S \subseteq TM$ in the codomain (the fact that $TM|_S$ is a submanifold of $TM$ follows for example from Lemma \ref{lem:pullback_of_bundle_under_embedding}).
\end{proof}

\begin{exercise}[Mentioned in \cite{bredon2013topology} in chapter 'Conventions for CW complexes']
    Let $(X,x_0)$ and $(Y,y_0)$ be compact pointed topological spaces, then the smash product $X \wedge Y$ is homeomorphic to the one point compactification of $Z := (X-x_0) \times (Y-y_0)$. In particular
    \[
    S^n \wedge S^m \cong S^{n+m}.
    \]
\end{exercise}
\begin{proof}[Proof sketch]
    Let $Z^+$ be the one point compactification of $Z$. Define $f \colon X \times Y \to Z^+$ by
    \[
    f(x,y) := \begin{dcases}
        (x,y) &, x \neq x_0 \text{ and } y \neq y_0 \\
        \infty &, \mathrm{otherwise}
    \end{dcases}
    \]
    It is then straightforward to show that $f$ is continuous and the following diagram is a push-out:
    \begin{center}
        \commsquare{$X \vee Y$}{$X \times Y$}{$\ast$}{$Z^+$}{}{}{$f$}{}
    \end{center} 
    So by the universal property of push-outs we get $Z^+ \cong X \wedge Y$. The statement about the smash product of spheres then just follows from $S^n - \ast \cong \R^n$ and that the one point compactification of $\R^n$ is $S^n$.
\end{proof}
