% Commutative Triangle
\newcommand{\commtriangle}[6]{
  \begin{tikzpicture}
    \draw node (node1) at (0,0) {#1};
    \draw node (node2) at (2,0) {#2};
    \draw node (node3) at (1,-1.5) {#3};

    \draw[->] (node1.east) to node[above] {#4} (node2.west);
    \draw[->] (node1.south) to node[above] {#5} ([xshift=-2]node3.north);
    \draw[->] (node2.south) to node[above] {#6} ([xshift=2]node3.north);
  \end{tikzpicture}
}

%Commutative Square
\newcommand{\commsquare}[8]{
  \begin{tikzpicture}
    \draw node (node1) at (0,0) {#1};
    \draw node (node2) at (2,0) {#2};
    \draw node (node3) at (0,-2) {#3};
    \draw node (node4) at (2,-2) {#4};

    \draw[->] (node1.east) to node[above] {#5} (node2.west);
    \draw[->] (node1.south) to node[right] {#6} (node3.north);
    \draw[->] (node2.south) to node[right] {#7} (node4.north);
    \draw[->] (node3.east) to node[above] {#8} (node4.west);
  \end{tikzpicture}
}
